public class Solution {
    public boolean containsDuplicate(int[] nums) {
        
        int length = nums.length;
        if(length == 0 || length == 1)
            return false;
        
        Set<Integer> set = new HashSet<Integer>();
        for(int i: nums){
            if(!set.add(i))
                return true;
        }
        
        return false;
    }
}
